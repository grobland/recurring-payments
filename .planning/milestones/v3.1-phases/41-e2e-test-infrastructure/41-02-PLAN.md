```typescript
    // Fill wrong email + wrong password
    // Click Sign in
    // Assert error message visible ("Invalid" or "incorrect" text)
    // Assert still on /login page
  });

  test("register page loads with form fields", async ({ page }) => {
    // Navigate to /register
    // Assert name, email, password fields visible
    // Assert "Create account" or "Sign up" button visible
    // DO NOT actually create an account (would pollute DB)
  });

  test("logout redirects to login page", async ({ page }) => {
    // This test DOES need auth state — but we set storageState to empty above
    // So first, log in via the UI:
    const email = process.env.TEST_USER_EMAIL!;
    const password = process.env.TEST_USER_PASSWORD!;
    await page.goto("/login");
    await page.locator("#email").fill(email);
    await page.locator("#password").fill(password);
    await page.getByRole("button", { name: "Sign in" }).click();
    await page.waitForURL("**/payments/dashboard**", { timeout: 15000 });

    // Now trigger logout via user menu
    await page.getByTestId("user-menu-trigger").click();
    await page.getByTestId("user-menu-logout").click();

    // Assert redirected to /login
    await expect(page).toHaveURL(/\/login/);
  });
});
```

Key implementation notes:
- Use `test.use({ storageState: { cookies: [], origins: [] } })` to override the default authenticated state.
- The logout test must first log in manually since it starts unauthenticated.
- Use `process.env.TEST_USER_EMAIL` and `process.env.TEST_USER_PASSWORD` for credentials.
- Use regex or glob for URL assertions to handle potential query params.
