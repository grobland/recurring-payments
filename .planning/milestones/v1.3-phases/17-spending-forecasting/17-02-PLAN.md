   ```typescript
   const today = new Date();
   const forecastEnd = addMonths(today, months);
   const monthlyForecasts: Record<string, number> = {};

   for (const sub of activeSubs) {
     let renewalDate = new Date(sub.nextRenewalDate);
     const incrementMonths = sub.frequency === 'monthly' ? 1 : 12;

     while (renewalDate <= forecastEnd) {
       if (renewalDate >= today) {
         const monthKey = format(renewalDate, 'yyyy-MM');
         monthlyForecasts[monthKey] =
           (monthlyForecasts[monthKey] || 0) +
           parseFloat(sub.normalizedMonthlyAmount);
       }
       renewalDate = addMonths(renewalDate, incrementMonths);
     }
   }
   ```

5. **Step 3 - Get historical data from MV for volatility**:
   ```typescript
   // Query past 12 months from user_analytics_mv
   const historicalData = await db.execute(sql`
     SELECT
       DATE_TRUNC('month', month) as month_start,
       SUM(normalized_monthly_amount) as total
     FROM user_analytics_mv
     WHERE user_id = ${session.user.id}
       AND month >= ${subMonths(today, 12)}
       AND month < ${today}
     GROUP BY DATE_TRUNC('month', month)
     ORDER BY month_start
   `);
   ```

3. **Step 1 - Calculate current monthly run rate**:
   - Query active subscriptions
   - Sum normalizedMonthlyAmount for current monthly spending

4. **Step 2 - Get historical data for volatility**:
   - Query user_analytics_mv for past 12 months (same as monthly endpoint)
   - Extract monthly totals array

5. **Step 3 - Calculate annual forecast with confidence intervals**:
   ```typescript
   const volatility = calculateVolatility(historicalMonthly);
   const monthlyStdDev = currentMonthlySpending * volatility;

   // Annual forecast = monthly * 12
   const annualForecast = currentMonthlySpending * 12;

   // Annual stdDev scales by sqrt(12) for 12 independent months
   const annualStdDev = monthlyStdDev * Math.sqrt(12);

   return {
     forecast: annualForecast,
     lower80: Math.max(0, annualForecast - 1.28 * annualStdDev),
     upper80: annualForecast + 1.28 * annualStdDev,
     lower95: Math.max(0, annualForecast - 1.96 * annualStdDev),
     upper95: annualForecast + 1.96 * annualStdDev,
   };
   ```

6. **Step 4 - Generate 12-month projections**:
   - Create monthlyProjections array with each month's forecast + CI
   - monthsAhead increments 1-12 for expanding bands
   - Use addConfidenceIntervals for each month

7. **Response**: Return AnnualForecastResponse with:
   - Annual totals with CI
   - monthlyProjections array (12 items)
   - metadata including monthlyStdDev

   export function useForecastCalendar(days: 30 | 60 | 90 = 30) {
     return useQuery({
       queryKey: forecastKeys.calendar(days),
       queryFn: () => fetchCalendar(days),
       staleTime: 5 * 60 * 1000, // 5 minutes
       refetchOnWindowFocus: false,
