import { db } from "@/lib/db";
import { stripePrices } from "@/lib/db/schema";
import type { Tier } from "@/lib/db/schema";

interface PriceConfig {
  stripePriceId: string;
  tier: Tier;
  interval: "month" | "year";
  currency: "usd" | "eur" | "gbp";
  amountCents: number;
}

// REPLACE THESE WITH ACTUAL PRICE IDs FROM STRIPE DASHBOARD
const PRICE_CONFIGS: PriceConfig[] = [
  // Primary tier - $4/€4/£3 monthly, $40/€40/£30 annual
  { stripePriceId: "price_PRIMARY_USD_MONTH", tier: "primary", interval: "month", currency: "usd", amountCents: 400 },
  { stripePriceId: "price_PRIMARY_USD_YEAR", tier: "primary", interval: "year", currency: "usd", amountCents: 4000 },
  { stripePriceId: "price_PRIMARY_EUR_MONTH", tier: "primary", interval: "month", currency: "eur", amountCents: 400 },
  { stripePriceId: "price_PRIMARY_EUR_YEAR", tier: "primary", interval: "year", currency: "eur", amountCents: 4000 },
  { stripePriceId: "price_PRIMARY_GBP_MONTH", tier: "primary", interval: "month", currency: "gbp", amountCents: 300 },
  { stripePriceId: "price_PRIMARY_GBP_YEAR", tier: "primary", interval: "year", currency: "gbp", amountCents: 3000 },

  // Enhanced tier - $7/€7/£5.50 monthly, $70/€70/£55 annual
  { stripePriceId: "price_ENHANCED_USD_MONTH", tier: "enhanced", interval: "month", currency: "usd", amountCents: 700 },
  { stripePriceId: "price_ENHANCED_USD_YEAR", tier: "enhanced", interval: "year", currency: "usd", amountCents: 7000 },
  { stripePriceId: "price_ENHANCED_EUR_MONTH", tier: "enhanced", interval: "month", currency: "eur", amountCents: 700 },
  { stripePriceId: "price_ENHANCED_EUR_YEAR", tier: "enhanced", interval: "year", currency: "eur", amountCents: 7000 },
  { stripePriceId: "price_ENHANCED_GBP_MONTH", tier: "enhanced", interval: "month", currency: "gbp", amountCents: 550 },
  { stripePriceId: "price_ENHANCED_GBP_YEAR", tier: "enhanced", interval: "year", currency: "gbp", amountCents: 5500 },

  // Advanced tier - $11/€11/£8.50 monthly, $110/€110/£85 annual
  { stripePriceId: "price_ADVANCED_USD_MONTH", tier: "advanced", interval: "month", currency: "usd", amountCents: 1100 },
  { stripePriceId: "price_ADVANCED_USD_YEAR", tier: "advanced", interval: "year", currency: "usd", amountCents: 11000 },
  { stripePriceId: "price_ADVANCED_EUR_MONTH", tier: "advanced", interval: "month", currency: "eur", amountCents: 1100 },
  { stripePriceId: "price_ADVANCED_EUR_YEAR", tier: "advanced", interval: "year", currency: "eur", amountCents: 11000 },
  { stripePriceId: "price_ADVANCED_GBP_MONTH", tier: "advanced", interval: "month", currency: "gbp", amountCents: 850 },
  { stripePriceId: "price_ADVANCED_GBP_YEAR", tier: "advanced", interval: "year", currency: "gbp", amountCents: 8500 },
];

async function seed() {
  console.log("Seeding stripe_prices table...");

  for (const config of PRICE_CONFIGS) {
    await db
      .insert(stripePrices)
      .values({
        stripePriceId: config.stripePriceId,
        tier: config.tier,
        interval: config.interval,
        currency: config.currency,
        amountCents: config.amountCents,
        isActive: true,
      })
      .onConflictDoUpdate({
        target: stripePrices.stripePriceId,
        set: {
          tier: config.tier,
          interval: config.interval,
          currency: config.currency,
          amountCents: config.amountCents,
          isActive: true,
        },
      });

    console.log(`  ✓ ${config.tier} ${config.currency.toUpperCase()} ${config.interval}`);
  }

  console.log(`\nSeeded ${PRICE_CONFIGS.length} prices successfully!`);
  process.exit(0);
}

seed().catch((error) => {
  console.error("Seed failed:", error);
  process.exit(1);
});

**Grandfathering Workflow Documentation**

The grandfathering mechanism is now built into the system through the `isActive` flag on stripe_prices records. When you need to update pricing:

**Workflow for Price Changes:**

1. **Create new prices in Stripe Dashboard**
   - Navigate to the product in Stripe Dashboard
   - Click "Add another price"
   - Enter new amounts for the tier/interval/currency you're updating
   - Save to get new price IDs

2. **Add new prices to database**
   - Run seed script with new price IDs added to PRICE_CONFIGS
   - Set `isActive: true` for new prices
   - Script uses `onConflictDoUpdate`, so re-running is safe

3. **Mark old prices as inactive**
   - Connect to database (via Drizzle Studio or SQL client)
   - Update old price records: `UPDATE stripe_prices SET is_active = false WHERE stripe_price_id = 'price_old_id'`
   - Do NOT delete old price records - users may still be subscribed to them

**Why this works:**
- `getPriceIdForCheckout()` only returns prices where `isActive = true` (new customers get new prices)
- `getGrandfatheringInfo()` compares user's current price to the latest active price for their tier
- Users with old prices remain on their existing subscription (grandfathered)
- The system automatically shows savings to grandfathered users

**Example:**
If Primary USD monthly was $4.00 and you raise it to $5.00:
1. Create new price in Stripe: `price_new123` at $5.00
2. Add to seed script: `{ stripePriceId: "price_new123", tier: "primary", interval: "month", currency: "usd", amountCents: 500, isActive: true }`
3. Run seed script
4. Update database: `UPDATE stripe_prices SET is_active = false WHERE stripe_price_id = 'price_old456'`
5. Existing users at $4.00 stay at $4.00, new users pay $5.00

**No code changes needed** - the tier derivation logic already handles this pattern through the `isActive` flag.
