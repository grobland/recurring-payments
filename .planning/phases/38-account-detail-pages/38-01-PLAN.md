Follow the exact algorithm from `src/app/api/vault/coverage/route.ts` but scoped to a single account.

```typescript

import { auth } from "@/lib/auth";
import { db } from "@/lib/db";
import { financialAccounts, statements } from "@/lib/db/schema";
import { eq, gte, and } from "drizzle-orm";
import { startOfMonth, subMonths, format } from "date-fns";

interface RouteParams {
  params: Promise<{ id: string }>;
}

export async function GET(request: Request, { params }: RouteParams) {
  // ... auth check (same as existing /api/accounts/[id]/route.ts)
  // const { id } = await params;  <-- MUST await params (Next.js 16)
  // Fetch account with userId ownership check
  // If account not found: return 404
  // If account has no linkedSourceType: return { sources: [], gapCount: 0, months: [...12 months] }
  //
  // Build 12-month window (identical to vault coverage):
  //   const now = new Date();
  //   const windowStart = startOfMonth(subMonths(now, 11));
  //   Build months[] array of 12 "yyyy-MM" strings
  //
  // Query statements WHERE:
  //   statements.userId = userId
  //   statements.accountId = id  (use account's id, NOT linkedSourceType string)
  //   statements.statementDate >= windowStart
  //
  // Build cellMap keyed by "sourceType::yyyy-MM" (same algorithm as vault coverage)
  //   - Prefer statement with PDF over one without for same cell
  //
  // Build sources array with exactly 1 entry (the linked source) with 12 cells
  // Count gaps (cells with state "missing")
  //
  // Return { sources, gapCount, months }
}
```

**CRITICAL:** Filter by `statements.accountId = id` (the FK), NOT by `statements.sourceType = account.linkedSourceType`. The FK is the authoritative link. This handles edge cases where a sourceType was relinked to a different account.

**CRITICAL:** The 12-month window computation must match vault coverage exactly — `startOfMonth(subMonths(now, 11))` for oldest month, same `format()` calls. `CoverageGrid` assumes `sources[n].cells.length === months.length === 12`.

If the account has no `linkedSourceType` AND no statements with `accountId = id`, return empty sources array with the 12 months array populated. This ensures `CoverageGrid` still receives valid month labels for header rendering.

"use client";

import { useQuery } from "@tanstack/react-query";
import type { CoverageResponse } from "@/lib/hooks/use-vault-coverage";

export const accountCoverageKeys = {
  coverage: (accountId: string) => ["accounts", accountId, "coverage"] as const,
};

async function fetchAccountCoverage(accountId: string): Promise<CoverageResponse> {
  const res = await fetch(`/api/accounts/${accountId}/coverage`);
  if (!res.ok) {
    const error = await res.json().catch(() => ({ error: "Failed to fetch account coverage" }));
    throw new Error(error.error || "Failed to fetch account coverage");

  return res.json();
}

export function useAccountCoverage(accountId: string) {
  return useQuery({
    queryKey: accountCoverageKeys.coverage(accountId),
    queryFn: () => fetchAccountCoverage(accountId),
    staleTime: 2 * 60 * 1000, // Match vault coverage staleTime
    enabled: !!accountId,
  });
}
